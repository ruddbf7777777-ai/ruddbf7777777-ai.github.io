  $tmp = intval($_GET['n']);

}

$prod = "1";
if (function_exists('bcscale')) bcscale(0);

$nums = [];

for ($i = 1; $i <= $n; $i++) {
  $nums[] = $i;
  $sum += $i;

  if (function_exists('bcmul')) {
    $prod = bcmul($prod, (string)$i);
  } else {
    $prod = (string)($prod * $i);
  }
}
