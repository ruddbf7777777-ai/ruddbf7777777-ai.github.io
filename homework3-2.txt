<?php
$n = 30;

if (isset($_GET['n'])) {
  $tmp = intval($_GET['n']);
  if ($tmp >= 10 && $tmp <= 100) $n = $tmp;
}

$arr = [];
$max = null;

for ($i = 0; $i < $n; $i++) {
  $v = rand(10, 100);
  $arr[] = $v;
  if ($max === null || $v > $max) $max = $v;
}

$sorted = $arr;
sort($sorted);
?>
